While loop

#include <stdio.h>

int main() {
    int x = 0;
    while (x <= 10) {
        printf("%i\n",x);
        x++;
    }
    while (x != 0) {
       if (x % 2 == 0) {
           printf("%i is divisible by 2\n",x);
       }
       x--;
    }
    return 0;
}
